95 void RealTimeInfoHook::DoGenerational() {
96  OrganismPoolPtr pool = GetMainPool();
97 
98  FitnessRecord record;
99  record.max = -INFINITY;
100  record.min = INFINITY;
101  record.mean = 0;
102  record.sd = 0;
103 
104  for (OrganismPtr org : *pool) {
105  double fitness = org->GetFitnessValue();
106  if (fitness > record.max)
107  record.max = fitness;
108  if (fitness < record.min)
109  record.min = fitness;
110  record.mean += fitness;
111  record.sd += fitness * fitness;
112  }
113 
114  record.mean /= pool->size();
115  record.sd = sqrt(record.sd / pool->size() - record.mean * record.mean);
116 
117  mFitnessRecords.push_back(record);
118 }

157 void RealTimeInfoHook::FitnessHandler(const Ptr<restbed::Session> session) {
158  const auto request = session->get_request();
159  llong from = 0, to = numeric_limits<llong>::max();
160  int res = EA_REALTIMEINFO_DEFAULT_RESOLUTION;
161 
162  // Parsing
163  try {
164  if (request->has_query_parameter("from"))
165  from = stoll(request->get_query_parameter("from"));
166  if (request->has_query_parameter("to"))
167  to = stoll(request->get_query_parameter("to"));
168  if (request->has_query_parameter("res"))
169  res = stoi(request->get_query_parameter("res"));
170  } catch (exception&) {
171  session->close(400, "");
172  return;
173  }
174 
175  // Validity check
176  if (from > to || from < 0 || to < 0 || res <= 0) {
177  session->close(400, "");
178  return;
179  }
180 
181  if (from < (llong)mStartGen) {
182  from = mStartGen;
183  if (to < from) to = from;
184  }
185  if (to >= (llong)(mStartGen + mFitnessRecords.size())) {
186  to = mStartGen + mFitnessRecords.size() - 1;
187  if (from > to) from = to;
188  }
189 
190  // Resolution calculation
191  llong delta = to - from;
192  if (delta + 1 < res)
193  res = delta + 1;
194 
195  int realRes = (res == 1 ? 1 : res - 1);
196  int step = delta / realRes;
197  double remain = (delta - (llong)step * realRes) / (double)realRes;
198 
199  // Put data in
200  ptree tree;
201 
202  double excess = 0;
203  llong gen = (res == 1 ? to : from);
204  for (int i = 0; i < res; i++, gen += step) {
205  ptree data;
206  llong realGen = gen + (excess >= 0.5 ? 1 : 0);
207  FitnessRecord record = mFitnessRecords[realGen - mStartGen];
208  data.put("max", record.max);
209  data.put("min", record.min);
210  data.put("mean", record.mean);
211  data.put("sd", record.sd);
212  tree.put_child(to_string(realGen), data);
213 
214  excess += remain;
215  if (excess >= 1) {
216  excess--;
217  gen++;
218  }
219  }
220 
221  // Response
222  Response(session, tree);
223 }
