131 void CMAEvolutionStrategy::Begin() {
132  weights = VectorXd::Constant(mMu, 1, log(mMu + 0.5)) - ArrayXd::LinSpaced(mMu, 1, mMu).log().matrix();
133  weights /= weights.sum();
134  mueff = 1 / weights.array().square().sum();
135 
136  cc = (4+mueff/N) / (N+4 + 2*mueff/N);
137  cs = (mueff+2) / (N+mueff+5);
138  c1 = 2 / (pow(N+1.3, 2) + mueff);
139  cmu = min(1-c1, 2 * (mueff-2 + 1/mueff) / (pow(N+2, 2) + mueff));
140  damps = 1 + 2*max(0., sqrt((mueff-1)/(N-1))-1) + cs;
141 
142  chiN = sqrt(N) * (1-1.0/(4*N)+1.0/(21*N*N));
143 }

145 void CMAEvolutionStrategy::Loop() {
146  mState = *GetPool(2)->To<CMAStatePool>();
147 
148  // Spawning
149  GenomePoolPtr spawnPool = make_shared<GenomePool>(mLambda);
150  Execute("S", function<void(void)>([&] () {
151  MultiThreading::For(0, mLambda, [&] (int i) {
152  VectorXd x = mean + sigma * B * (D.array() * RandomVector().array()).matrix();
153  DoubleArrayGenomePtr genome = make_shared<DoubleArrayGenome>();
154  genome->GetGenes() = vector<double>(x.data(), x.data() + x.size());
155  (*spawnPool)[i] = static_pointer_cast<Genome>(genome);
156  });
157  }));
158 
159  // Evaluation
160  OrganismPoolPtr evaluatedPool = Execute("E", evaluator, spawnPool);
161  evaluatedPool->Sort();
162 
163  // Extract
164  OrganismPoolPtr filteredPool = GreedySelection().Select(evaluatedPool, mMu);
165  MatrixXd X(N, mMu);
166  MultiThreading::For(0, mMu, [&] (int i) {
167  double* begin = &(*filteredPool->at(i)->GetGenome()->To<DoubleArrayGenome>()->GetGenes().begin());
168  X.col(i) = Map<VectorXd>(begin, N);
169  });
170 
171  // Update
172  MatrixXd C;
173  Execute("U", function<void(void)>([&] () {
174  VectorXd meanOld = mean;
175  mean = X * weights;
176 
177  ps = (1-cs)*ps + sqrt(cs*(2-cs)*mueff) * mState.GetInvSqrtC() * (mean - meanOld) / sigma;
178  auto counteval = GetPopulation()->GetEvaluation();
179  bool hsig = ps.norm() / sqrt(1.0 - pow(1-cs, 2.0*counteval/mLambda))/chiN < 1.4 + 2.0/(N+1);
180  pc = (1-cc)*pc;
181  if (hsig)
182  pc += sqrt(cc*(2-cc)*mueff) * (mean - meanOld) / sigma;
183 
184  MatrixXd Xtmp = (1/sigma) * (X.colwise() - meanOld);
185  C = mState.GetC();
186  C = (1-c1-cmu) * C +
187  c1 * (pc * pc.transpose() + (hsig ? 0 : 1) * cc * (2-cc) * C) +
188  cmu * Xtmp * weights.asDiagonal() * Xtmp.transpose();
189 
190  sigma *= exp((cs/damps)*(ps.norm()/chiN - 1));
191  }));
192 
193  Execute("D", function<void(void)>(bind(&CMAEvolutionStrategy::Decompose, this, ref(C))));
194 
195  // Finalize
196  SetPool(0, evaluatedPool);
197  SetPool(2, make_shared<CMAStatePool>(mState));
198 }
