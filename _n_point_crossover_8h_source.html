96 GenomePtr NPointCrossover<T>::DoCombine(vector<GenomePtr>& pParents) {
97  vector<vector<T>*> genomes(pParents.size());
98  transform(pParents.begin(), pParents.end(), genomes.begin(),
99  [this](GenomePtr parent) {
100  ArrayGenomePtr<T> genome = dynamic_pointer_cast<ArrayGenome<T>>(parent);
101  if (!genome)
102  throw invalid_argument(
103  "All input parents for NPointCrossover must be ArrayGenome-s.");
104  if (genome->GetSize() <= mCrossCount)
105  throw invalid_argument(
106  "All input parents for " + to_string(mCrossCount) +
107  "-PointCrossover must have at least " + to_string(mCrossCount + 1) + " genes.");
108  return &genome->GetGenes();
109  });
110 
111  uint minSize = min((*genomes[0]).size(), (*genomes[1]).size());
112  vector<uint> crossPoints;
113  if (mCrossCount == 1) {
114  uniform_int_distribution<uint> dist(1, minSize - 1);
115  crossPoints = {0, dist(Random::generator)};
116  }
117  else {
118  crossPoints = vector<uint>(minSize - 1);
119  iota(crossPoints.begin(), crossPoints.end(), 1);
120  shuffle(crossPoints.begin(), crossPoints.end(), Random::generator);
121  crossPoints.erase(crossPoints.begin() + mCrossCount, crossPoints.end());
122  sort(crossPoints.begin(), crossPoints.end());
123  crossPoints.insert(crossPoints.begin(), 0);
124  }
125 
126  ArrayGenomePtr<T> newGenome = make_shared<ArrayGenome<T>>();
127  vector<T>& newGenes = newGenome->GetGenes();
128  uint parent = 0;
129  for (uint i = 0; i < mCrossCount; i++) {
130  newGenes.insert(newGenes.end(),
131  (*genomes[parent]).begin() + crossPoints[i],
132  (*genomes[parent]).begin() + crossPoints[i + 1]);
133 
134  parent ^= 1;
135  }
136  newGenes.insert(newGenes.end(),
137  (*genomes[parent]).begin() + crossPoints[mCrossCount],
138  (*genomes[parent]).end());
139 
140  return newGenome;
141 }
