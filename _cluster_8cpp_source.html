159 void Cluster::SlaveRoutine() {
160  signal(SIGINT, SIG_IGN);
161  signal(SIGTERM, SIG_IGN);
162  Log::Redirect(Log::TRACE, "stdout");
163  Log::Redirect(Log::DEBUG, "stdout");
164 
165  int rank;
166  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
167 
168  char name[100];
169  int nameLen;
170  MPI_Get_processor_name(name, &nameLen);
171  string pc(name, nameLen);
172  EA_LOG_TRACE << "Cluster Slave node #" + to_string(rank) + " on " + pc + " started" << flush;
173 
174  while (true) {
175  MPI_Status status;
176  MPI_Probe(0, MPI_ANY_TAG, MPI_COMM_WORLD, &status);
177 
178  int tag = status.MPI_TAG;
179 
180  // Terminate
181  if (tag == 0) {
182  char tmp;
183  MPI_Recv(&tmp, 1, MPI_BYTE, 0, tag, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
184  break;
185  }
186  // Load function
187  else if (tag == 1) {
188  int opId;
189  MPI_Recv(&opId, 1, MPI_INT, 0, tag, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
190  sCurrentOp = sOperators[opId];
191  MPI_Barrier(MPI_COMM_WORLD);
192  continue;
193  }
194  // Unload function
195  else if (tag == 2) {
196  char tmp;
197  MPI_Recv(&tmp, 1, MPI_BYTE, 0, tag, MPI_COMM_WORLD, MPI_STATUS_IGNORE);
198  sCurrentOp = { };
199  MPI_Barrier(MPI_COMM_WORLD);
200  continue;
201  }
202 
203  if (!sCurrentOp)
204  throw EA_EXCEPTION(EAException, CLUSTER_OPERATOR_NOT_LOADED,
205  "Cluster operator have not been loaded in cluster slave node #" + to_string(rank));
206 
207  int count;
208  MPI_Get_count(&status, MPI_BYTE, &count);
209 
210  sCurrentOp(count, tag);
211  }
212 
213  // End protocol
214  MPI_Finalize();

216  exit(0);
