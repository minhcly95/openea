63 void BackupHook::DoInitial() {
64  using namespace boost::filesystem;
65 
66  path p(mDir);
67  if (exists(p)) {
68  if (is_directory(p)) {
69  if (mClear) {
70  long long generation = GetGeneration();
71  uint fileCleared = 0;
72  for (directory_entry& entry : directory_iterator(p)) {
73  path entryPath = entry.path();
74  if (is_regular_file(entryPath)
75  && ExtractGenerationNumber(entryPath.string())
76  > generation) {
77  remove(entryPath);
78  fileCleared++;
79  }
80  }
81  EA_LOG_DEBUG<< "BackupHook::DoInitial: Cleared " << fileCleared << " file(s) in \"" << mDir << "\"." << flush;
82  }
83  } else
84  throw EA_EXCEPTION(EAException, EXISTED_FILE_IS_NOT_DIRECTORY,
85  "BackupHook::DoInitial: \"" + mDir + "\" exists but is not a directory.");
86  } else {
87  if (create_directories(p))
88  EA_LOG_DEBUG << "BackupHook::DoInitial: Directory \"" + mDir + "\" created." << flush;
89  else
90  throw EA_EXCEPTION(EAException, CANNOT_CREATE_FILE,
91  "BackupHook::DoInitial: Cannot create directory \"" + mDir + "\".");
92  }
