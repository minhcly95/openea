104  const VariableMap& map) {

108  for (const XMLAttribute* attr = elem->FirstAttribute(); attr;

111  forward_as_tuple(AttrValue(attr->Value(), map)));

113 
114  for (const XMLElement* child = elem->FirstChildElement(); child; child = child->NextSiblingElement()) {
115  vector<ConstructiblePtr> objList;

118  objList.push_back(Construct(obj, map));
119  }
120 
121  if (objList.size() == 1)

123  forward_as_tuple(objList[0]));

126  forward_as_tuple(objList));
127  }

129  ConstructiblePtr obj = NameService::Get(className).Construct(dataMap);
130  mConstructed.push_back(obj);

134 string EAMLReader::AttrValue(string str, const VariableMap& map) {
135  vector<string> tokens;
136  boost::split(tokens, str, boost::is_any_of("|"));
137 
138  string input = tokens[0];
